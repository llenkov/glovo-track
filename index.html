// ── DASHBOARD ── async function loadDash() { const weeks = await getAllWeeks(); const allExp = await getAllExpenses(); const allPay = await getAllPayments(); const el = document.getElementById('dash'); if (!weeks.length) { el.innerHTML = `
📭

Няма въведена седмица.
Добави първата от меню → Нова седмица.

`; return; } // Очаквани брутни приходи (не получени) const expectedGross = weeks .filter(w => !w.is_received) .reduce((a, b) => a + b.gross_income, 0); // Очаквани нетени приходи (не получени) const expectedNet = weeks .filter(w => !w.is_received) .reduce((a, b) => a + b.net_income, 0); // Получени брутни приходи const receivedGross = weeks .filter(w => w.is_received) .reduce((a, b) => a + b.gross_income, 0); // Всички разходи const totalExpenses = allExp.reduce((a, b) => a + b.amount, 0); // Всички получени плащания const totalPayments = allPay.reduce((a, b) => a + b.amount, 0); // Налични пари: Плащания - Разходи const availableMoney = +(totalPayments - totalExpenses).toFixed(2); const isAvailablePositive = availableMoney >= 0; el.innerHTML = `
📅 Всички периоди
Брутен приход (Очаквани)
€${expectedGross.toFixed(2)}
Нетен: €${expectedNet.toFixed(2)}
Брутен приход (Получени)
€${receivedGross.toFixed(2)}
Маркирани като получени
Всички разходи
€${totalExpenses.toFixed(2)}
${allExp.length} записа
Всички плащания
€${totalPayments.toFixed(2)}
${allPay.length} плащания
Налични пари
€${Math.abs(availableMoney).toFixed(2)}
${isAvailablePositive?'✓ Налично':'✗ Дефицит'} · Плащания €${totalPayments.toFixed(2)} − Разходи €${totalExpenses.toFixed(2)}
`; }
